  /**
  * @param {boolean[][]} grid
  * @returns {number[][]}

  function detectBombs(grid) {
    const bombsGrid = []
    for (const i in grid) {
      bombsGrid[i] = []
      for (const j in grid[i]) {
        bombsGrid[i][j] = 
          ~~grid[i - 1]?.[j - 1] +
